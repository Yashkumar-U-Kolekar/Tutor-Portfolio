   245→  backImage = null,
   246→  imageFit = 'cover',
   247→  lanyardImage = null,

   264→  // isn't supplied for a given face, then skip compositing it below.

   266→  const backTex = useTexture(backImage || BLANK_PIXEL);
   267→
   268→  // Composite the front/back images into the card's texture atlas (front = left
   269→  // half, back = right half). Each image is drawn aspect-preserving (no stretch).
   270→  const cardMap = useMemo(() => {
   271→    const baseMap = materials.base.map;
   272→    if (!frontImage && !backImage) return baseMap;
   273→
   274→    const baseImg = baseMap.image;
   275→    const W = baseImg.width;
   276→    const H = baseImg.height;
   277→    const canvas = document.createElement('canvas');
   278→    canvas.width = W;
   279→    canvas.height = H;
   280→    const ctx = canvas.getContext('2d');
   281→    if (!ctx) return baseMap;
   282→    // Keep the original baked atlas for the card edges and any untouched face.
   283→    ctx.drawImage(baseImg, 0, 0, W, H);
   284→
   285→    const drawFitted = (img, rect) => {
   286→      const rx = rect.x * W;
   287→      const ry = rect.y * H;
   288→      const rw = rect.w * W;
   289→      const rh = rect.h * H;
   290→      const pick = imageFit === 'contain' ? Math.min : Math.max;
   291→      const scale = pick(rw / img.width, rh / img.height);
   292→      const dw = img.width * scale;
   293→      const dh = img.height * scale;
   294→      const dx = rx + (rw - dw) / 2;
   295→      const dy = ry + (rh - dh) / 2;
   296→      ctx.save();
   297→      ctx.beginPath();
   298→      ctx.rect(rx, ry, rw, rh);
   299→      ctx.clip();
   300→      ctx.drawImage(img, dx, dy, dw, dh);
   301→      ctx.restore();
   302→    };
   303→
   304→    if (frontImage && frontTex.image) drawFitted(frontTex.image, FRONT_UV_RECT);
   305→    if (backImage && backTex.image) drawFitted(backTex.image, BACK_UV_RECT);
   306→
   307→    const composite = new THREE.CanvasTexture(canvas);
   308→    composite.colorSpace = THREE.SRGBColorSpace;
   309→    composite.flipY = baseMap.flipY;
   310→    composite.anisotropy = 16;
   311→    composite.needsUpdate = true;
   312→    return composite;

   325→    [0, 0, 0],
   326→    [0, 1.5, 0]
   327→  ]);

   331→      document.body.style.cursor = dragged ? 'grabbing' : 'grab';
   332→      return () => void (document.body.style.cursor = 'auto');
   333→    }
   334→  }, [hovered, dragged]);
   335→

   340→      vec.add(dir.multiplyScalar(state.camera.position.length()));

   352→      });

   354→      curve.points[1].copy(j2.current.lerped);

   530→  behindGlowColor,
   531→  behindGlowSize,
   532→  className = '',
   533→  enableTilt = true,
   534→  enableMobileTilt = false,
   535→  mobileTiltSensitivity = 5,
   536→  miniAvatarUrl,
   537→  name = 'Javi A. Torres',
   538→  title = 'Software Engineer',
   539→  handle = 'javicodes',
   540→  status = 'Online',
   541→  contactText = 'Contact',
   542→  showUserInfo = true,
   543→  onContactClick
   544→}) => {
   545→  const wrapRef = useRef(null);
   546→  const shellRef = useRef(null);
   547→
   548→  const enterTimerRef = useRef(null);
   549→  const leaveRafRef = useRef(null);
   550→
   551→  const tiltEngine = useMemo(() => {
   552→    if (!enableTilt) return null;
   553→
   554→    let rafId = null;
   555→    let running = false;
   556→    let lastTs = 0;
   557→
   558→    let currentX = 0;
   559→    let currentY = 0;
   560→    let targetX = 0;
   561→    let targetY = 0;
   562→
   563→    const DEFAULT_TAU = 0.14;
   564→    const INITIAL_TAU = 0.6;
   565→    let initialUntil = 0;
   566→
   567→    const setVarsFromXY = (x, y) => {
   568→      const shell = shellRef.current;
   569→      const wrap = wrapRef.current;
   570→      if (!shell || !wrap) return;
   571→
   572→      const width = shell.clientWidth || 1;
   573→      const height = shell.clientHeight || 1;
   574→
   575→      const percentX = clamp((100 / width) * x);
   576→      const percentY = clamp((100 / height) * y);
   577→
   578→      const centerX = percentX - 50;
   579→      const centerY = percentY - 50;
   580→
   581→      const properties = {
   582→        '--pointer-x': `${percentX}%`,
   583→        '--pointer-y': `${percentY}%`,
   584→        '--background-x': `${adjust(percentX, 0, 100, 35, 65)}%`,
   585→        '--background-y': `${adjust(percentY, 0, 100, 35, 65)}%`,
   586→        '--pointer-from-center': `${clamp(Math.hypot(percentY - 50, percentX - 50) / 50, 0, 1)}`,
   587→        '--pointer-from-top': `${percentY / 100}`,
   588→        '--pointer-from-left': `${percentX / 100}`,
   589→        '--rotate-x': `${round(-(centerX / 5))}deg`,
   590→        '--rotate-y': `${round(centerY / 4)}deg`
   591→      };
   592→
   593→      for (const [k, v] of Object.entries(properties)) wrap.style.setProperty(k, v);
   594→    };
   595→
   596→    const step = ts => {
   597→      if (!running) return;
   598→      if (lastTs === 0) lastTs = ts;
   599→      const dt = (ts - lastTs) / 1000;
   600→      lastTs = ts;
   601→
   602→      const tau = ts < initialUntil ? INITIAL_TAU : DEFAULT_TAU;
   603→      const k = 1 - Math.exp(-dt / tau);
   604→
   605→      currentX += (targetX - currentX) * k;
   606→      currentY += (targetY - currentY) * k;
   607→
   608→      setVarsFromXY(currentX, currentY);
   609→
   610→      const stillFar = Math.abs(targetX - currentX) > 0.05 || Math.abs(targetY - currentY) > 0.05;
   611→
   612→      if (stillFar || document.hasFocus()) {
   613→        rafId = requestAnimationFrame(step);
   614→      } else {
   615→        running = false;
   616→        lastTs = 0;
   617→        if (rafId) {
   618→          cancelAnimationFrame(rafId);
   619→          rafId = null;
   620→        }
   621→      }
   622→    };
   623→
   624→    const start = () => {
   625→      if (running) return;
   626→      running = true;
   627→      lastTs = 0;
   628→      rafId = requestAnimationFrame(step);
   629→    };
   630→
   631→    return {
   632→      setImmediate(x, y) {
   633→        currentX = x;
   634→        currentY = y;
   635→        setVarsFromXY(currentX, currentY);
   636→      },
   637→      setTarget(x, y) {
   638→        targetX = x;
   639→        targetY = y;
   640→        start();
   641→      },
   642→      toCenter() {
   643→        const shell = shellRef.current;
   644→        if (!shell) return;
   645→        this.setTarget(shell.clientWidth / 2, shell.clientHeight / 2);
   646→      },
   647→      beginInitial(durationMs) {
   648→        initialUntil = performance.now() + durationMs;
   649→        start();
   650→      },
   651→      getCurrent() {
   652→        return { x: currentX, y: currentY, tx: targetX, ty: targetY };
   653→      },
   654→      cancel() {
   655→        if (rafId) cancelAnimationFrame(rafId);
   656→        rafId = null;
   657→        running = false;
   658→        lastTs = 0;
   659→      }
   660→    };
   661→  }, [enableTilt]);
   662→
   663→  const getOffsets = (evt, el) => {
   664→    const rect = el.getBoundingClientRect();
   665→    return { x: evt.clientX - rect.left, y: evt.clientY - rect.top };
   666→  };
   667→
   668→  const handlePointerMove = useCallback(
   669→    event => {
   670→      const shell = shellRef.current;
   671→      if (!shell || !tiltEngine) return;
   672→      const { x, y } = getOffsets(event, shell);
   673→      tiltEngine.setTarget(x, y);
   674→    },
   675→    [tiltEngine]
   676→  );
   677→
   678→  const handlePointerEnter = useCallback(
   679→    event => {
   680→      const shell = shellRef.current;
   681→      if (!shell || !tiltEngine) return;
   682→
   683→      shell.classList.add('active');
   684→      shell.classList.add('entering');
   685→      if (enterTimerRef.current) window.clearTimeout(enterTimerRef.current);
   686→      enterTimerRef.current = window.setTimeout(() => {
   687→        shell.classList.remove('entering');
   688→      }, ANIMATION_CONFIG.ENTER_TRANSITION_MS);
   689→
   690→      const { x, y } = getOffsets(event, shell);
   691→      tiltEngine.setTarget(x, y);
   692→    },
   693→    [tiltEngine]
   694→  );
   695→
   696→  const handlePointerLeave = useCallback(() => {
   697→    const shell = shellRef.current;
   698→    if (!shell || !tiltEngine) return;
   699→
   700→    tiltEngine.toCenter();
   701→
   702→    const checkSettle = () => {
   703→      const { x, y, tx, ty } = tiltEngine.getCurrent();
   704→      const settled = Math.hypot(tx - x, ty - y) < 0.6;
   705→      if (settled) {
   706→        shell.classList.remove('active');
   707→        leaveRafRef.current = null;
   708→      } else {
   709→        leaveRafRef.current = requestAnimationFrame(checkSettle);
   710→      }
   711→    };
   712→    if (leaveRafRef.current) cancelAnimationFrame(leaveRafRef.current);
   713→    leaveRafRef.current = requestAnimationFrame(checkSettle);
   714→  }, [tiltEngine]);
   715→
   716→  const handleDeviceOrientation = useCallback(
   717→    event => {
   718→      const shell = shellRef.current;
   719→      if (!shell || !tiltEngine) return;
   720→
   721→      const { beta, gamma } = event;
   722→      if (beta == null || gamma == null) return;
   723→
   724→      const centerX = shell.clientWidth / 2;
   725→      const centerY = shell.clientHeight / 2;
   726→      const x = clamp(centerX + gamma * mobileTiltSensitivity, 0, shell.clientWidth);
   727→      const y = clamp(
   728→        centerY + (beta - ANIMATION_CONFIG.DEVICE_BETA_OFFSET) * mobileTiltSensitivity,
   729→        0,
   730→        shell.clientHeight
   731→      );
   732→
   733→      tiltEngine.setTarget(x, y);
   734→    },
   735→    [tiltEngine, mobileTiltSensitivity]
   736→  );
   737→
   738→  useEffect(() => {
   739→    if (!enableTilt || !tiltEngine) return;
   740→
   741→    const shell = shellRef.current;
   742→    if (!shell) return;
   743→
   744→    const pointerMoveHandler = handlePointerMove;
   745→    const pointerEnterHandler = handlePointerEnter;
   746→    const pointerLeaveHandler = handlePointerLeave;
   747→    const deviceOrientationHandler = handleDeviceOrientation;
   748→
   749→    shell.addEventListener('pointerenter', pointerEnterHandler);
   750→    shell.addEventListener('pointermove', pointerMoveHandler);
   751→    shell.addEventListener('pointerleave', pointerLeaveHandler);
   752→
   753→    const handleClick = () => {
   754→      if (!enableMobileTilt || location.protocol !== 'https:') return;
   755→      const anyMotion = window.DeviceMotionEvent;
   756→      if (anyMotion && typeof anyMotion.requestPermission === 'function') {
   757→        anyMotion
   758→          .requestPermission()
   759→          .then(state => {
   760→            if (state === 'granted') {
   761→              window.addEventListener('deviceorientation', deviceOrientationHandler);
   762→            }
   763→          })
   764→          .catch(console.error);
   765→      } else {
   766→        window.addEventListener('deviceorientation', deviceOrientationHandler);
   767→      }
   768→    };
   769→    shell.addEventListener('click', handleClick);
   770→
   771→    const initialX = (shell.clientWidth || 0) - ANIMATION_CONFIG.INITIAL_X_OFFSET;
   772→    const initialY = ANIMATION_CONFIG.INITIAL_Y_OFFSET;
   773→    tiltEngine.setImmediate(initialX, initialY);
   774→    tiltEngine.toCenter();
   775→    tiltEngine.beginInitial(ANIMATION_CONFIG.INITIAL_DURATION);
   776→
   777→    return () => {
   778→      shell.removeEventListener('pointerenter', pointerEnterHandler);
   779→      shell.removeEventListener('pointermove', pointerMoveHandler);
   780→      shell.removeEventListener('pointerleave', pointerLeaveHandler);
   781→      shell.removeEventListener('click', handleClick);
   782→      window.removeEventListener('deviceorientation', deviceOrientationHandler);
   783→      if (enterTimerRef.current) window.clearTimeout(enterTimerRef.current);
   784→      if (leaveRafRef.current) cancelAnimationFrame(leaveRafRef.current);
   785→      tiltEngine.cancel();
   786→      shell.classList.remove('entering');
   787→    };
   788→  }, [
   789→    enableTilt,
   790→    enableMobileTilt,
   791→    tiltEngine,
   792→    handlePointerMove,
   793→    handlePointerEnter,
   794→    handlePointerLeave,
   795→    handleDeviceOrientation
   796→  ]);
   797→
   798→  const cardStyle = useMemo(
   799→    () => ({
   800→      '--icon': iconUrl ? `url(${iconUrl})` : 'none',
   801→      '--grain': grainUrl ? `url(${grainUrl})` : 'none',
   802→      '--inner-gradient': innerGradient ?? DEFAULT_INNER_GRADIENT,
   803→      '--behind-glow-color': behindGlowColor ?? 'rgba(125, 190, 255, 0.67)',
   804→      '--behind-glow-size': behindGlowSize ?? '50%'
   805→    }),
   806→    [iconUrl, grainUrl, innerGradient, behindGlowColor, behindGlowSize]
   807→  );
   808→
   809→  const handleContactClick = useCallback(() => {
   810→    onContactClick?.();
   811→  }, [onContactClick]);
   812→
   813→  return (
   814→    <div ref={wrapRef} className={`pc-card-wrapper ${className}`.trim()} style={cardStyle}>
   815→      {behindGlowEnabled && <div className="pc-behind" />}
   816→      <div ref={shellRef} className="pc-card-shell">
   817→        <section className="pc-card">
   818→          <div className="pc-inside">
   819→            <div className="pc-shine" />
   820→            <div className="pc-glare" />
   821→            <div className="pc-content pc-avatar-content">
   822→              <img
   823→                className="avatar"
   824→                src={avatarUrl}
   825→                alt={`${name || 'User'} avatar`}
   826→                loading="lazy"
   827→                onError={e => {
   828→                  const t = e.target;
   829→                  t.style.display = 'none';
   830→                }}
   831→              />
   832→              {showUserInfo && (
   833→                <div className="pc-user-info">
   834→                  <div className="pc-user-details">
   835→                    <div className="pc-mini-avatar">
   836→                      <img
   837→                        src={miniAvatarUrl || avatarUrl}
   838→                        alt={`${name || 'User'} mini avatar`}
   839→                        loading="lazy"
   840→                        onError={e => {
   841→                          const t = e.target;
   842→                          t.style.opacity = '0.5';
   843→                          t.src = avatarUrl;
   844→                        }}
   845→                      />
   846→                    </div>
   847→                    <div className="pc-user-text">
   848→                      <div className="pc-handle">@{handle}</div>
   849→                      <div className="pc-status">{status}</div>
   850→                    </div>
   851→                  </div>
   852→                  <button
   853→                    className="pc-contact-btn"
   854→                    onClick={handleContactClick}
   855→                    style={{ pointerEvents: 'auto' }}
   856→                    type="button"
   857→                    aria-label={`Contact ${name || 'user'}`}
   858→                  >
   859→                    {contactText}
   860→                  </button>
   861→                </div>
   862→              )}
   863→            </div>
   864→            <div className="pc-content">
   865→              <div className="pc-details">
   866→                <h3>{name}</h3>
   867→                <p>{title}</p>
   868→              </div>
   869→            </div>
   870→          </div>
   871→        </section>
   872→      </div>
   873→    </div>
   874→  );
   875→};
   876→
   877→const ProfileCard = React.memo(ProfileCardComponent);
   878→export default ProfileCard;

   682тЖТ   682тЖТ

   692тЖТ   692тЖТ    },
   693тЖТ   693тЖТ    [tiltEngine]
   694тЖТ   694тЖТ  );
   695тЖТ   695тЖТ
   696тЖТ   696тЖТ  const handlePointerLeave = useCallback(() => {
   697тЖТ   697тЖТ    const shell = shellRef.current;
   698тЖТ   698тЖТ    if (!shell || !tiltEngine) return;
   699тЖТ   699тЖТ
   700тЖТ   700тЖТ    tiltEngine.toCenter();

   702тЖТ   702тЖТ    const checkSettle = () => {
   703тЖТ   703тЖТ      const { x, y, tx, ty } = tiltEngine.getCurrent();

   705тЖТ   705тЖТ      if (settled) {
   706тЖТ   706тЖТ        shell.classList.remove('active');

   708тЖТ   708тЖТ      } else {
   709тЖТ   709тЖТ        leaveRafRef.current = requestAnimationFrame(checkSettle);
   710тЖТ   710тЖТ      }
   711тЖТ   711тЖТ    };

   713тЖТ   713тЖТ    leaveRafRef.current = requestAnimationFrame(checkSettle);
   714тЖТ   714тЖТ  }, [tiltEngine]);
   715тЖТ   715тЖТ
   716тЖТ   716тЖТ  const handleDeviceOrientation = useCallback(
   717тЖТ   717тЖТ    event => {
   718тЖТ   718тЖТ      const shell = shellRef.current;

   720тЖТ   720тЖТ
   721тЖТ   721тЖТ      const { beta, gamma } = event;

   725тЖТ   725тЖТ      const centerY = shell.clientHeight / 2;

   727тЖТ   727тЖТ      const y = clamp(
   728тЖТ   728тЖТ        centerY + (beta - ANIMATION_CONFIG.DEVICE_BETA_OFFSET) * mobileTiltSensitivity,
   729тЖТ   729тЖТ        0,
   730тЖТ   730тЖТ        shell.clientHeight
   731тЖТ   731тЖТ      );
   732тЖТ   732тЖТ
   733тЖТ   733тЖТ      tiltEngine.setTarget(x, y);
   734тЖТ   734тЖТ    },
   735тЖТ   735тЖТ    [tiltEngine, mobileTiltSensitivity]
   736тЖТ   736тЖТ  );
   737тЖТ   737тЖТ
   738тЖТ   738тЖТ  useEffect(() => {

   750тЖТ   750тЖТ    shell.addEventListener('pointermove', pointerMoveHandler);
   751тЖТ   751тЖТ    shell.addEventListener('pointerleave', pointerLeaveHandler);
   752тЖТ   752тЖТ

   756тЖТ   756тЖТ      if (anyMotion && typeof anyMotion.requestPermission === 'function') {

   762тЖТ   762тЖТ            }
   763тЖТ   763тЖТ          })
   764тЖТ   764тЖТ          .catch(console.error);
   765тЖТ   765тЖТ      } else {
   766тЖТ   766тЖТ        window.addEventListener('deviceorientation', deviceOrientationHandler);
   767тЖТ   767тЖТ      }
   768тЖТ   768тЖТ    };
   769тЖТ   769тЖТ    shell.addEventListener('click', handleClick);
   770тЖТ   770тЖТ
   771тЖТ   771тЖТ    const initialX = (shell.clientWidth || 0) - ANIMATION_CONFIG.INITIAL_X_OFFSET;
   772тЖТ   772тЖТ    const initialY = ANIMATION_CONFIG.INITIAL_Y_OFFSET;

   779тЖТ   779тЖТ      shell.removeEventListener('pointermove', pointerMoveHandler);
   780тЖТ   780тЖТ      shell.removeEventListener('pointerleave', pointerLeaveHandler);
   781тЖТ   781тЖТ      shell.removeEventListener('click', handleClick);
   782тЖТ   782тЖТ      window.removeEventListener('deviceorientation', deviceOrientationHandler);
   783тЖТ   783тЖТ      if (enterTimerRef.current) window.clearTimeout(enterTimerRef.current);
   784тЖТ   784тЖТ      if (leaveRafRef.current) cancelAnimationFrame(leaveRafRef.current);
   785тЖТ   785тЖТ      tiltEngine.cancel();
   786тЖТ   786тЖТ      shell.classList.remove('entering');
   787тЖТ   787тЖТ    };
   788тЖТ   788тЖТ  }, [
   789тЖТ   789тЖТ    enableTilt,
   790тЖТ   790тЖТ    enableMobileTilt,
   791тЖТ   791тЖТ    tiltEngine,
   792тЖТ   792тЖТ    handlePointerMove,
   793тЖТ   793тЖТ    handlePointerEnter,
   794тЖТ   794тЖТ    handlePointerLeave,
   795тЖТ   795тЖТ    handleDeviceOrientation
   796тЖТ   796тЖТ  ]);
   797тЖТ   797тЖТ
   798тЖТ   798тЖТ  const cardStyle = useMemo(
   799тЖТ   799тЖТ    () => ({
   800тЖТ   800тЖТ      '--icon': iconUrl ? `url(${iconUrl})` : 'none',

   806тЖТ   806тЖТ    [iconUrl, grainUrl, innerGradient, behindGlowColor, behindGlowSize]
   807тЖТ   807тЖТ  );
   808тЖТ   808тЖТ
   809тЖТ   809тЖТ  const handleContactClick = useCallback(() => {
   810тЖТ   810тЖТ    onContactClick?.();
   811тЖТ   811тЖТ  }, [onContactClick]);
   812тЖТ   812тЖТ
   813тЖТ   813тЖТ  return (
